# Object Card

A composable, two-panel card for displaying entity detail overlays (Ship, Person, Concept, Voyage, Letter). The left panel shows structured properties, the right panel shows references. Overlay/modal positioning is the consumer's responsibility — this component renders only the card itself.

## Usage

<CodeBlock code={`import {
  ObjectCard,
  ObjectCardBadge,
  ObjectCardBody,
  ObjectCardExternalLink,
  ObjectCardFooter,
  ObjectCardFooterAction,
  ObjectCardHeader,
  ObjectCardListItem,
  ObjectCardPanel,
  ObjectCardProperty,
  ObjectCardPropertyList,
  ObjectCardReferenceItem,
  ObjectCardSection,
  ObjectCardStat,
  ObjectCardStats,
  ObjectCardTitle,
  ObjectCardViewMore,
  IconCopy,
  IconDownload,
} from "@globalise/design-system";

<ObjectCard>
  <ObjectCardHeader onClose={() => {}}>
    <ObjectCardBadge type="ship">Ship</ObjectCardBadge>
    <ObjectCardTitle>Prins Eugenius</ObjectCardTitle>
    <ObjectCardStats>
      <ObjectCardStat>7 Voyages</ObjectCardStat>
      <ObjectCardStat>1,234 References</ObjectCardStat>
    </ObjectCardStats>
  </ObjectCardHeader>

  <ObjectCardBody className="h-[600px]">
    <ObjectCardPanel side="left" className="gap-12">
      <ObjectCardSection title="General Properties">
        <ObjectCardPropertyList>
          <ObjectCardProperty label="Built" value="1703, Amsterdam" />
          <ObjectCardProperty label="Weight" value="874 tons" />
        </ObjectCardPropertyList>
      </ObjectCardSection>

      <ObjectCardSection title="External Identifiers">
        <div className="flex flex-col gap-3">
          <ObjectCardExternalLink href="https://example.com">
            DAS ship1203
          </ObjectCardExternalLink>
        </div>
      </ObjectCardSection>

      <ObjectCardFooter>
        <ObjectCardFooterAction icon={<IconCopy className="h-3.5 w-3.5" />}>
          Copy URI
        </ObjectCardFooterAction>
      </ObjectCardFooter>
    </ObjectCardPanel>

    <ObjectCardPanel side="right">
      <span className="font-sans text-sm text-zinc-400">
        References (1,234)
      </span>
      <ObjectCardReferenceItem
        title="p.264"
        snippet="in 't geheel p:r de prins Eugenius..."
        archiveId="NL-HaNA 1.04.02 · 10070_0054 · 264"
        href="#"
      />
    </ObjectCardPanel>

  </ObjectCardBody>
</ObjectCard>`} />
