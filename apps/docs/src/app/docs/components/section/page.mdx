# Section & SectionDivider

Layout primitives for dividing pages into background‑coloured bands with consistent vertical spacing, and inserting a styled divider between them.

## Preview

<ComponentPreview>
  <div className="w-full overflow-hidden border border-white/10">
    <Section background="dark" spacing="default">
      <Container>
        <Typography variant="h3" className="text-white">
          Dark section (default spacing)
        </Typography>
        <Typography variant="p" className="text-white/60">
          Background: brand‑black
        </Typography>
      </Container>
    </Section>
    <Section background="light" spacing="default">
      <Container>
        <Typography variant="h3">Light section</Typography>
        <Typography variant="p" className="text-black/60">
          Background: white
        </Typography>
      </Container>
    </Section>
  </div>
</ComponentPreview>

## Usage

<CodeBlock code={`import { Section, SectionDivider, Container, Typography } from "@globalise/design-system";

<Section background="dark" spacing="large">
  <Container>
    <Typography variant="h1" className="text-white">
      Hero
    </Typography>
  </Container>
</Section>

<SectionDivider />

<Section background="dark">
  <Container>
    <Typography variant="h2" className="text-white">Content</Typography>
  </Container>
</Section>`} />
